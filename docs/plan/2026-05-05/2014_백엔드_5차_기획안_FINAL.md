## 1. Member 모델 (Django 표준 패턴)

### 1-1. 코드

```python
# apps/accounts/models.py
from django.contrib.auth.models import AbstractBaseUser, BaseUserManager, PermissionsMixin
from django.db import models


class MemberManager(BaseUserManager):
    def create_user(self, id, email, name, phone, password=None, **extra):
        if not id:
            raise ValueError('ID는 필수입니다.')
        if not email:
            raise ValueError('이메일은 필수입니다.')
        user = self.model(
            id=id,
            email=self.normalize_email(email),
            name=name,
            phone=phone,
            **extra,
        )
        user.set_password(password)  # 자동 해싱 (PBKDF2 기본)
        user.save(using=self._db)
        return user

    def create_superuser(self, id, email, name, phone, password, **extra):
        extra.setdefault('role', 'admin')
        extra.setdefault('is_staff', True)
        extra.setdefault('is_superuser', True)
        return self.create_user(id, email, name, phone, password, **extra)


class Member(AbstractBaseUser, PermissionsMixin):
    """
    회원. 사용자가 가입 시 직접 ID 정함 (영문/숫자/_, 3~20자, 영구 불변).
    AbstractBaseUser 가 password (해싱), is_active, last_login 자동 제공.
    PermissionsMixin 이 is_superuser, groups, user_permissions 자동 제공.
    """
    ROLE_CHOICES = [('member', '회원'), ('admin', '어드민')]
    PROVIDER_CHOICES = [('email', '이메일'), ('kakao', '카카오'), ('google', '구글')]

    id           = models.SlugField(primary_key=True, max_length=20)         # 'smk6931'
    email        = models.EmailField(unique=True)                            # 가입 시 중복 검증
    name         = models.CharField(max_length=50)                           # 실명 (입금자 확인용)
    phone        = models.CharField(max_length=20)                           # 전화번호
    age          = models.PositiveSmallIntegerField(null=True, blank=True)   # 선택
    display_name = models.CharField(max_length=50, blank=True)               # 한국어 표시명 (선택, '손명관')
    role         = models.CharField(max_length=10, choices=ROLE_CHOICES, default='member')
    provider     = models.CharField(max_length=10, choices=PROVIDER_CHOICES, default='email')
    provider_id  = models.CharField(max_length=100, blank=True)              # 카카오/구글 OAuth ID
    avatar_url   = models.URLField(blank=True)
    bio          = models.TextField(blank=True)

    # AbstractBaseUser 가 자동 제공: password, last_login
    # PermissionsMixin 이 자동 제공: is_superuser, groups, user_permissions
    is_staff   = models.BooleanField(default=False)
    is_active  = models.BooleanField(default=True)   # 휴면

    created_at = models.DateTimeField(auto_now_add=True)
    updated_at = models.DateTimeField(auto_now=True)

    objects = MemberManager()

    USERNAME_FIELD  = 'id'                           # 로그인 시 ID 로
    REQUIRED_FIELDS = ['email', 'name', 'phone']     # createsuperuser 시 필수

    class Meta:
        db_table = 'member'

    def __str__(self):
        return f'{self.id} ({self.name})'

    @property
    def display(self):
        """UI 표시: display_name 있으면 그것, 없으면 id"""
        return self.display_name or self.id
```

### 10-2. handle 자동 생성

```python
def generate_handle(email):
    base = email.split('@')[0]                  # 'smk6931@gmail.com' → 'smk6931'
    base = re.sub(r'[^a-z0-9_]', '_', base.lower())[:18]
    candidate, suffix = base, 1
    while Member.objects.filter(handle=candidate).exists():
        suffix += 1
        candidate = f'{base}_{suffix}'[:20]
    return candidate
```
